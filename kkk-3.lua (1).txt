local function updateWalkAnimation()
	if humanoid.MoveDirection.Magnitude > 0 then
		if not walkAnimation.IsPlaying then
			walkAnimation:Play(walkFadeTime)
			walkAnimation.TimePosition = 0
		end
		walkAnimation:AdjustSpeed(walkAnimationBaseSpeed)
	else
		if walkAnimation.IsPlaying then
			walkAnimation:Stop(walkFadeTime)
		end
	end
end

local function playAnimation(animId, fade, timePos, speed, stopDelay)
	local anim = Instance.new("Animation")
	anim.AnimationId = animId

	local track
	local success, err = pcall(function()
		track = animator:LoadAnimation(anim)
	end)

	if success and track then
		track:Play(fade)
		track.TimePosition = timePos
		track:AdjustSpeed(speed)

		task.delay(stopDelay, function()
			if track and track.IsPlaying then
				track:Stop(fade)
			end
		end)
		return track
	end
	return nil
end

local function screenShake(duration, intensity)
	local startTime = tick()
	local shakeConnection
	shakeConnection = RunService.RenderStepped:Connect(function()
		local elapsed = tick() - startTime
		if elapsed >= duration or not camera then
			shakeConnection:Disconnect()
			return
		end
		local offset = Vector3.new(
			math.random(-100, 100) / 100 * intensity,
			math.random(-100, 100) / 100 * intensity,
			0
		)
		camera.CFrame = camera.CFrame * CFrame.new(offset)
	end)
end

local function playPunchEffect(loops, interval, volume, duration, intensity, effectColor)

	task.spawn(function()
		local sound = Instance.new("Sound")
		sound.SoundId = "rbxassetid://139697578472716"

		sound.Parent = camera

		for i = 1, loops do
			sound:Play()
			screenShake(duration, intensity)

			spawnImpactBurst(hitCFrame, effectColor, 10)

		end
		sound:Destroy()
	end)
end

local function lerp(a, b, t)
	return a + (b - a) * t
end

local function lerpAngle(a, b, t)
	return a + math.atan2(math.sin(b - a), math.cos(b - a)) * t
end

local function applyPush(force, getActiveFlag)
	task.spawn(function()
		if not ReplicatedStorage:FindFirstChild("push") then
			local detection = Instance.new("Decal")
			detection.Name = "push"
			detection.Parent = ReplicatedStorage
		end
		local lp = Players.LocalPlayer
		local movel = 0.1
		while humanoid.Health > 0 do
			local active = getActiveFlag()
			if not active then
				break
			end
			RunService.Heartbeat:Wait()
			local c = lp.Character
			local pushHrp = c and c:FindFirstChild("HumanoidRootPart")
			if pushHrp then
				local vel = pushHrp.Velocity
				pushHrp.Velocity = pushHrp.CFrame.LookVector * force + Vector3.new(0, force, 0)
				RunService.RenderStepped:Wait()
				if pushHrp then pushHrp.Velocity = vel end
				RunService.Stepped:Wait()
				if pushHrp then 
					pushHrp.Velocity = vel + Vector3.new(0, movel, 0)
				end
				movel = -movel
			end
		end
	end)
end

local function executeM1()
	movementActive = false
	humanoid.AutoRotate = false

	local attachment = Instance.new("Attachment")
	attachment.Parent = hrp

	local align = Instance.new("AlignOrientation")
	align.Mode = Enum.OrientationAlignmentMode.OneAttachment
	align.Attachment0 = attachment
	align.MaxTorque = math.huge
	align.Responsiveness = 200
	align.RigidityEnabled = true
	align.Parent = hrp

	local Frames = {
		{ Time = 0.5, RX = 35, RY = 30, RZ = 0 },
		{ Time = 0.3, RX = -25, RY = -30, RZ = 0 },
		{ Time = 0.2, RX = -30, RY = -35, RZ = 0 },
		{ Time = 0.25, RX = 0, RY = 0, RZ = 0 }
	}

	playAnimation("rbxassetid://233329237", 0.3, 1, -0.19, 1)
	playAnimation("rbxassetid://21633130", 0.3, 0, 0.1, 1)
	playPunchEffect(1, 0.8, 6, 0.25, 0.35)

	local pushActive = true
	applyPush(190, function() return pushActive end)

	task.delay(0.85, function()
		pushActive = false
	end)

	local currentYaw = math.rad(hrp.Orientation.Y)
	local frame = 1
	local frameStart = tick()
	local done = false

	local connection
	connection = RunService.Heartbeat:Connect(function(dt)
		if humanoid.Health <= 0 or not align or not attachment then
			connection:Disconnect()
			done = true
			return
		end

		local moveDir = humanoid.MoveDirection
		local targetYaw = currentYaw
		if moveDir.Magnitude > 0.01 then
			targetYaw = math.atan2(-moveDir.X, -moveDir.Z)
		end
		currentYaw = lerpAngle(currentYaw, targetYaw, dt * 15)

		local current = Frames[frame]
		if not current then
			connection:Disconnect()
			done = true
			return
		end

		local previous = Frames[frame - 1]
		local startRX = previous and previous.RX or 0
		local startRY = previous and previous.RY or 0
		local startRZ = previous and previous.RZ or 0

		local alpha = math.clamp((tick() - frameStart) / current.Time, 0, 1)
		local rx = lerp(startRX, current.RX, alpha)
		local ry = lerp(startRY, current.RY, alpha)
		local rz = lerp(startRZ, current.RZ, alpha)

		align.CFrame = CFrame.Angles(0, currentYaw, 0) * CFrame.Angles(math.rad(rx), math.rad(ry), math.rad(rz))

		if alpha >= 1 then

RunService.Heartbeat:Connect(function(dt)
	if humanoid.Health <= 0 then return end
	
	if not movementActive then
		if galeTrack and galeTrack.IsPlaying then
			galeTrack:Stop(0.3)
		end
		wasGaleMoving = false
		galeStopping = false
		sprintTrail.Enabled = false
		return
	end

	local velocity = (hrp.Position - lastPos) / math.max(dt, 1 / 60)
	lastPos = hrp.Position
	local planarSpeed = Vector2.new(velocity.X, velocity.Z).Magnitude
	local moving = planarSpeed > 0.1

	if galeTrack then
		if isGaleRunning and moving then
			if not galeTrack.IsPlaying then
				galeTrack:Play(0.3)
				galeTrack.TimePosition = 0.2
				galeTrack:AdjustSpeed(0)
			end
		else
			if galeTrack.IsPlaying then
				galeTrack:Stop(0.3)
			end
		end
	end

	sprintTrail.Enabled = isGaleRunning and moving

	if isGaleRunning and moving then
		if not wasGaleMoving then
			wasGaleMoving = true
			galeStopping = false
			galeFrame = 1
			galeFrameStart = tick()
		end

		local current = WalkTiltFrames[galeFrame]
		local prevIndex = galeFrame - 1
		if prevIndex < 1 then prevIndex = #WalkTiltFrames end
		local previous = WalkTiltFrames[prevIndex]

		local alpha = math.clamp((tick() - galeFrameStart) / current.Time, 0, 1)
		galeRX = lerp(previous.RX, current.RX, alpha)
		galeRY = lerp(previous.RY, current.RY, alpha)
		galeRZ = lerp(previous.RZ, current.RZ, alpha)

		if alpha >= 1 then
			galeFrame += 1
			if galeFrame > #WalkTiltFrames then galeFrame = 1 end
			galeFrameStart = tick()
		end
	elseif wasGaleMoving then
		wasGaleMoving = false
		galeStopping = true
		galeStopStart = tick()
		galeLastRX, galeLastRY, galeLastRZ = galeRX, galeRY, galeRZ
	end

	if galeStopping then
		local alpha = math.clamp((tick() - galeStopStart) / 0.3, 0, 1)
		galeRX = lerp(galeLastRX, 0, alpha)
		galeRY = lerp(galeLastRY, 0, alpha)
		galeRZ = lerp(galeLastRZ, 0, alpha)
		if alpha >= 1 then galeStopping = false end
	elseif not (isGaleRunning and moving) then
		galeRX, galeRY, galeRZ = 0, 0, 0
	end

	local targetSlope = 0
	local ray = Workspace:Raycast(hrp.Position, Vector3.new(0, -10, 0), rayParams)

	if ray and ray.Instance and ray.Instance.CanCollide then
		targetSlope = -hrp.CFrame.LookVector:Dot(ray.Normal) * SLOPE_MAX * 1.5
	end

	if humanoid.Sit then targetSlope *= 2 end
	slopeTilt += (targetSlope - slopeTilt) * dt * SMOOTH

	local targetX, targetZ = 0, 0
	local moveDir = humanoid.MoveDirection
	local targetIdleX, targetIdleY, targetIdleZ = 0, 0, 0

	if IdleConfig.Enabled and planarSpeed < IdleConfig.IdleThreshold then
		targetIdleX = IdleConfig.RotationX
		targetIdleY = IdleConfig.RotationY
		targetIdleZ = IdleConfig.RotationZ
	end

	idleX += (targetIdleX - idleX) * dt * IdleConfig.Smooth
	idleY += (targetIdleY - idleY) * dt * IdleConfig.Smooth
	idleZ += (targetIdleZ - idleZ) * dt * IdleConfig.Smooth

	local baseTilt = humanoid.Sit and SIT_TILT_MAX or MOVE_TILT_MAX

	if planarSpeed > 0.1 then
		local forwardDot = moveDir:Dot(hrp.CFrame.LookVector)
		local moveMultiplier = (planarSpeed < 16) and LeanConfig.WalkLean or LeanConfig.RunLean
		if forwardDot < 0 then moveMultiplier *= LeanConfig.BackwardsLean end

		targetX = -forwardDot * baseTilt * moveMultiplier
		local rightDot = moveDir:Dot(hrp.CFrame.RightVector)
		if rightDot > 0 then
			targetZ = -rightDot * baseTilt * LeanConfig.RightLean
		else
			targetZ = -rightDot * baseTilt * LeanConfig.LeftLean
		end
	end

	local intensity = math.clamp(velocity.Y / 50, -1, 1)
	local targetY
	if velocity.Y > 0 then
		targetY = intensity * VERTICAL_UP_TILT * LeanConfig.JumpLean
	else
		targetY = intensity * VERTICAL_DOWN_TILT * LeanConfig.FallLean
	end

	if humanoid.Sit then targetY *= 2 end

	tiltX += (targetX - tiltX) * dt * SMOOTH
	tiltY += (targetY - tiltY) * dt * SMOOTH
	tiltZ += (targetZ - tiltZ) * dt * SMOOTH

	if WalkSwayConfig.Enabled and not humanoid.Sit then
		gaitPhase += (planarSpeed * dt) / WalkSwayConfig.StepLength
		local swayIntensity = math.clamp((planarSpeed - WalkSwayConfig.MinSpeed) / (WalkSwayConfig.FullSpeed - WalkSwayConfig.MinSpeed), 0, 1)
		local wave = math.sin(gaitPhase * math.pi * 2)

		local targetSwayZ = wave * WalkSwayConfig.Amount * swayIntensity
		local targetSwayY = wave * WalkSwayConfig.YawAmount * swayIntensity
		local targetBob = math.abs(wave) * WalkSwayConfig.BobAmount * swayIntensity
		local targetHip = defaultHipHeight - (math.abs(wave) * WalkSwayConfig.HipBob * swayIntensity)

		swayZ += (targetSwayZ - swayZ) * dt * WalkSwayConfig.Smooth
		swayY += (targetSwayY - swayY) * dt * WalkSwayConfig.Smooth
		bobOffset += (targetBob - bobOffset) * dt * WalkSwayConfig.Smooth

		currentHipHeight += (targetHip - currentHipHeight) * dt * WalkSwayConfig.HipSmooth
		humanoid.HipHeight = currentHipHeight
	else
		swayZ += (0 - swayZ) * dt * WalkSwayConfig.Smooth
		swayY += (0 - swayY) * dt * WalkSwayConfig.Smooth
		bobOffset += (0 - bobOffset) * dt * WalkSwayConfig.Smooth

		currentHipHeight += (defaultHipHeight - currentHipHeight) * dt * WalkSwayConfig.HipSmooth
		humanoid.HipHeight = currentHipHeight
	end

	local targetYaw = currentYaw
	if isShiftLock then
		local cam = Workspace.CurrentCamera
		if cam then targetYaw = math.atan2(-cam.CFrame.LookVector.X, -cam.CFrame.LookVector.Z) end
	elseif moveDir.Magnitude > 0.1 then
		targetYaw = math.atan2(-moveDir.X, -moveDir.Z)
	end

	currentYaw = lerpAngle(currentYaw, targetYaw, dt * (isShiftLock and 40 or 10))
	moveAttachment.Position = Vector3.new(0, bobOffset, 0)

	if isGaleRunning and (moving or galeStopping) then
		moveAlign.CFrame =
			CFrame.Angles(0, currentYaw + swayY, 0) *
			CFrame.Angles(math.rad(galeRX) + (tiltY + slopeTilt + (math.sin(gaitPhase * math.pi * 2) * WalkSwayConfig.PitchAmount)), math.rad(galeRY), math.rad(galeRZ)) *
			CFrame.Angles(tiltX, 0, tiltZ + swayZ) *
			CFrame.Angles(idleX, idleY, idleZ)
	else
		moveAlign.CFrame =
			CFrame.Angles(0, currentYaw + swayY, 0) *
			CFrame.Angles(tiltY + slopeTilt + (math.sin(gaitPhase * math.pi * 2) * WalkSwayConfig.PitchAmount), 0, 0) *
			CFrame.Angles(tiltX, 0, tiltZ + swayZ) *
			CFrame.Angles(idleX, idleY, idleZ)
	end
end)

local function applyButtonEffects(button)
	local uiCorner = Instance.new("UICorner")
	uiCorner.CornerRadius = UDim.new(0.15, 0)
	uiCorner.Parent = button

	local originalSize = button.Size
	local shrinkSize = UDim2.new(
		originalSize.X.Scale, originalSize.X.Offset * 0.85,
		originalSize.Y.Scale, originalSize.Y.Offset * 0.85
	)

	local tweenInfo = TweenInfo.new(0.1, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
	local shrinkTween = TweenService:Create(button, tweenInfo, {Size = shrinkSize})
	local growTween = TweenService:Create(button, tweenInfo, {Size = originalSize})

	button.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			shrinkTween:Play()
		end
	end)

	button.InputEnded:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			growTween:Play()
		end
	end)

	button.MouseLeave:Connect(function()
		growTween:Play()
	end)
end

local function handlePunch()
	if isAttacking or punchCooldown or humanoid.Health <= 0 then return end
	isAttacking = true
	lastPunchTime = tick()

	if isGaleRunning then
		executeM3()
		isAttacking = false

		punchCooldown = true
		punchButton.ImageColor3 = Color3.fromRGB(100, 100, 100)
		local cooldownStart = tick()
		while tick() - cooldownStart < 0.7 and humanoid.Health > 0 do
			punchLabel.Text = string.format("%.1fs", 0.7 - (tick() - cooldownStart))
			task.wait()
		end
		punchLabel.Text = ""
		punchButton.ImageColor3 = Color3.fromRGB(255, 255, 255)
		punchCooldown = false
	else
		if comboIndex == 1 then
			executeM1()
			if humanoid.Health > 0 then comboIndex = 2 end
			isAttacking = false
		elseif comboIndex == 2 then
			executeM2()
			if humanoid.Health > 0 then comboIndex = 3 end
			isAttacking = false
		elseif comboIndex == 3 then
			executeM3()
			comboIndex = 1
			isAttacking = false

			punchCooldown = true
			punchButton.ImageColor3 = Color3.fromRGB(100, 100, 100)
			local cooldownStart = tick()
			while tick() - cooldownStart < 0.7 and humanoid.Health > 0 do
				punchLabel.Text = string.format("%.1fs", 0.7 - (tick() - cooldownStart))
				task.wait()
			end
			punchLabel.Text = ""
			punchButton.ImageColor3 = Color3.fromRGB(255, 255, 255)
			punchCooldown = false
		end
	end
end

		Size = Vector3.new(0.2, 14, 14),
		Transparency = 1,
	}):Play()

		chunk.Anchored = false
		chunk.CanCollide = true
		chunk.CanQuery = false

			math.random(60, 90),

		)

			math.random(4, 9) / 10,
			math.random(4, 9) / 10,
			math.random(4, 9) / 10
		)

			math.random(-10, 10) / 10,
			math.random(0, 5) / 10,
			math.random(-10, 10) / 10
		) * CFrame.Angles(math.random(0, 360), math.random(0, 360), math.random(0, 360))
		chunk.Parent = Workspace

		local dir = Vector3.new(
			math.random(-100, 100),
			math.random(50, 130),
			math.random(-100, 100)
		)
		if dir.Magnitude == 0 then dir = Vector3.new(0, 1, 0) end
		dir = dir.Unit

			math.random(-14, 14),
			math.random(-14, 14)

	dust.SpreadAngle = Vector2.new(180, 180)

	dust.Parent = dustPart

local function triggerSprintLandingBoost()
	if sprintLandBoostActive or humanoid.Health <= 0 then return end
	sprintLandBoostActive = true
	sprintBoostToken += 1
	local myToken = sprintBoostToken

	local targetSpeed = math.min(
		(isGaleRunning and galeRunSpeed or baseWalkSpeed) * sprintLandBoostMultiplier,
		sprintLandBoostMaxSpeed
	)
	humanoid.WalkSpeed = targetSpeed

	spawnSonicBoomEffect(hrp.CFrame * CFrame.new(0, -2.6, 0), hrp.CFrame.LookVector)

	local kick = Instance.new("BodyVelocity")
	kick.Velocity = hrp.CFrame.LookVector * (targetSpeed * 0.6)
	kick.MaxForce = Vector3.new(40000, 0, 40000)
	kick.P = 3000
	kick.Parent = hrp
	Debris:AddItem(kick, 0.2)

	local boostTrail, boostTrailCleanup = attachTrail(
		hrp,
		Color3.fromRGB(255, 255, 255),
		Vector3.new(0, 0.9, 0),
		Vector3.new(0, -0.9, 0),
		0.3,
		NumberSequence.new({
			NumberSequenceKeypoint.new(0, 0.6),
			NumberSequenceKeypoint.new(1, 0),
		})
	)
	sprintBoostTrailCleanup = boostTrailCleanup

	task.delay(sprintLandBoostDuration, function()
		if sprintBoostToken ~= myToken then return end
		if boostTrailCleanup then boostTrailCleanup() end
		sprintBoostTrailCleanup = nil
		if humanoid.Health > 0 then
			humanoid.WalkSpeed = isGaleRunning and galeRunSpeed or baseWalkSpeed
		end
		sprintLandBoostActive = false
	end)
end

local function handleGaleJump()
	if isAttacking or jumpCooldown or humanoid.Health <= 0 then return end
	isAttacking = true
	jumpCooldown = true
	jumpButtonUI.ImageColor3 = Color3.fromRGB(100, 100, 100)
	movementActive = false
	humanoid.AutoRotate = false

	local wasSprintingIntoJump = isGaleRunning

	playAnimation("rbxassetid://97170520", 0.5, 0, 1, 3)

	local leapDistance = 20
	local leapHeight = 80

	spawnRing(hrp.Position - Vector3.new(0, 3, 0), Color3.fromRGB(220, 220, 220), 1, 9, 0.4)
	spawnImpactBurst(hrp.CFrame * CFrame.new(0, -2.5, 0), Color3.fromRGB(230, 230, 230), 14, 0.8)
	spawnDebris(hrp.CFrame * CFrame.new(0, -2.5, 0), Color3.fromRGB(200, 200, 205), 6)
	local jumpTrail, jumpTrailCleanup = attachTrail(
		hrp,
		Color3.fromRGB(235, 240, 255),
		Vector3.new(0, 1.5, 0),
		Vector3.new(0, -1.5, 0),
		0.4
	)
	task.delay(1, jumpTrailCleanup)

	local bodyVelocity = Instance.new("BodyVelocity")
	bodyVelocity.Velocity = hrp.CFrame.LookVector * leapDistance + Vector3.new(0, leapHeight, 0)
	bodyVelocity.MaxForce = Vector3.new(100000, 100000, 100000)
	bodyVelocity.P = 5000
	bodyVelocity.Parent = hrp
	Debris:AddItem(bodyVelocity, 0.1)

	local attachment = Instance.new("Attachment")
	attachment.Parent = hrp

	local align = Instance.new("AlignOrientation")
	align.Mode = Enum.OrientationAlignmentMode.OneAttachment
	align.Attachment0 = attachment
	align.MaxTorque = math.huge
	align.Responsiveness = 200
	align.RigidityEnabled = true
	align.Parent = hrp

	local Frames = {
		{ Time = 0.5, RX = 40, RY = -30, RZ = 0 },
		{ Time = 0.4, RX = -25, RY = 0, RZ = 0 },
		{ Time = 0.2, RX = -30, RY = 1, RZ = 0 },
		{ Time = 0.3, RX = 0, RY = 0, RZ = 0 }
	}

	local currentYaw = math.rad(hrp.Orientation.Y)
	local frame = 1
	local frameStart = tick()
	local done = false
	local jumpStartTime = tick()

	local groundCheckParams = RaycastParams.new()
	groundCheckParams.FilterType = Enum.RaycastFilterType.Exclude
	groundCheckParams.FilterDescendantsInstances = {character}

	local connection
	connection = RunService.Heartbeat:Connect(function(dt)
		if humanoid.Health <= 0 or not align or not attachment then
			connection:Disconnect()
			done = true
			return
		end

		local moveDir = humanoid.MoveDirection
		local targetYaw = currentYaw

		if moveDir.Magnitude > 0.01 then
			targetYaw = math.atan2(-moveDir.X, -moveDir.Z)
		end

		currentYaw = lerpAngle(currentYaw, targetYaw, dt * 15)
		local current = Frames[frame]

		if current then
			local previous = Frames[frame - 1]
			local startRX = previous and previous.RX or 0
			local startRY = previous and previous.RY or 0
			local startRZ = previous and previous.RZ or 0

			local alpha = math.clamp((tick() - frameStart) / current.Time, 0, 1)
			local rx = lerp(startRX, current.RX, alpha)
			local ry = lerp(startRY, current.RY, alpha)
			local rz = lerp(startRZ, current.RZ, alpha)

			align.CFrame = CFrame.Angles(0, currentYaw, 0) * CFrame.Angles(math.rad(rx), math.rad(ry), math.rad(rz))

			if alpha >= 1 then
				frame += 1
				frameStart = tick()
			end
		else
			local last = Frames[#Frames]
			align.CFrame = CFrame.Angles(0, currentYaw, 0) * CFrame.Angles(math.rad(last.RX), math.rad(last.RY), math.rad(last.RZ))
		end

		if tick() - jumpStartTime > 0.15 then
			local ray = Workspace:Raycast(hrp.Position, Vector3.new(0, -3.6, 0), groundCheckParams)
			if ray and ray.Instance and ray.Instance.CanCollide and hrp.AssemblyLinearVelocity.Y <= 0.5 then
				connection:Disconnect()
				done = true
			end
		end
	end)

	while not done do task.wait() end

	if align then align:Destroy() end
	if attachment then attachment:Destroy() end
	spawnRing(hrp.Position - Vector3.new(0, 3, 0), Color3.fromRGB(220, 220, 220), 1, 8, 0.35)
	spawnDebris(hrp.CFrame * CFrame.new(0, -2.5, 0), Color3.fromRGB(200, 200, 205), 6)
	screenFlash(Color3.fromRGB(230, 230, 235), 0.85, 0.15)
	cameraPunch(3, 0.15)

	if wasSprintingIntoJump then
		triggerSprintLandingBoost()
	end

	humanoid.AutoRotate = true
	movementActive = true
	isAttacking = false

	local cooldownDuration = 1.5
	local cooldownStart = tick()
	while tick() - cooldownStart < cooldownDuration and humanoid.Health > 0 do
		jumpLabel.Text = string.format("%.1fs", cooldownDuration - (tick() - cooldownStart))
		task.wait()
	end
	jumpLabel.Text = ""
	jumpButtonUI.ImageColor3 = Color3.fromRGB(255, 255, 255)
	jumpCooldown = false
end

local function toggleGaleRun()
	if humanoid.Health <= 0 then return end
	isGaleRunning = not isGaleRunning

	if isGaleRunning then
		if not sprintLandBoostActive then
			humanoid.WalkSpeed = galeRunSpeed
		end
		galeButton.ImageColor3 = Color3.fromRGB(255, 255, 255)
	else
		endSprintLandingBoost()
		humanoid.WalkSpeed = baseWalkSpeed
		galeButton.ImageColor3 = Color3.fromRGB(100, 100, 100)
	end
end

	end

local dragging = false
local dragInput
local dragStart
local startPos

local function update(input)
	local delta = input.Position - dragStart
	antiflingFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end

antiflingFrame.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
		dragging = true
		dragStart = input.Position
		startPos = antiflingFrame.Position

		input.Changed:Connect(function()
			if input.UserInputState == Enum.UserInputState.End then
				dragging = false
			end
		end)
	end
end)

antiflingFrame.InputChanged:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
		dragInput = input
	end
end)

UserInputService.InputChanged:Connect(function(input)
	if input == dragInput and dragging then
		update(input)
	end
end)
